	INSERT INTO GQUERIES VALUES ('GERA_GWUSER','GERA GVIEWS','
	DECLARE @COD_USER VARCHAR(20)
	DECLARE @COD_CURRENT_USER VARCHAR(20)

	SET @COD_USER = ''#COD_USER''
	SET @COD_CURRENT_USER = ''#COD_CURRENT_USER''

	DELETE FROM GWVIEWS WHERE COD_CURRENT_USER = @COD_CURRENT_USER
	DELETE FROM GWVIEWUSE WHERE COD_CURRENT_USER = @COD_CURRENT_USER

	INSERT INTO GWVIEWS 
	SELECT @COD_CURRENT_USER, @COD_USER, V.CODIGO, NOME, CASE WHEN ISNULL(VIEWUSER.RECNUM,0) <> 0 THEN VIEWUSER.STATUS ELSE 0 END  FROM GVIEWS V
		OUTER APPLY (SELECT * FROM GVIEWUSER VU WHERE VU.COD_VIEW = V.CODIGO AND VU.COD_USER = @COD_USER) AS VIEWUSER

	INSERT INTO GWVIEWUSE
	SELECT @COD_CURRENT_USER, @COD_USER, COD_VIEW, GVIEWS.NOME, STATUS FROM GVIEWUSER
		INNER JOIN GVIEWS ON GVIEWS.CODIGO = GVIEWUSER.COD_VIEW
	WHERE GVIEWUSER.COD_USER = @COD_USER')
